Self-portrait

This is my self-portrait that I made through p5.js!

function setup() {
createCanvas(600, 600);
}

function draw() {
background('#C9C9C9');

fill(0,0,0);
rect(180,180,250,250);


fill(204,108,231);
ellipse(300,475,200,320);


fill(231,219,201);
ellipse(300,300,200,200);

fill(106,71,59);
circle(250,250,45,45);
circle(350,250,45,45);

fill(255,255,255);
circle(240,250,20,20);
circle(340,250,20,20);


line(250,300,300,250);
line(250,300,325,300);
line(260,320,310,320);

}
